        function handleCSVUpload(e) {

            const file = e.target.files[0];

            if (!file) return;


            Papa.parse(file, {

                header: false,

                skipEmptyLines: true,

                complete: function(results) {

                    processCSVData(results.data);

                    // Reset input

                    e.target.value = '';

                },

                error: function(err) {

                    alert("เกิดข้อผิดพลาดในการอ่านไฟล์ CSV: " + err.message);

                }

            });

        }


        function processCSVData(rows) {

            let currentProject = "General Project";

            let parsedData = [];

            let idCounter = 1;


            // ตรวจสอบโครงสร้างว่าบรรทัดแรกเป็น Header ไหม ถ้าใช่ให้ข้าม

            let startIndex = 0;

            if (rows[0] && rows[0].join('').toLowerCase().includes('business unit')) {

                startIndex = 1;

            }


            for (let i = startIndex; i < rows.length; i++) {

                let row = rows[i];

                

                // ถ้าคอลัมน์แรกมีข้อมูล แต่คอลัมน์อื่นๆ ว่างหมด -> ถือว่าเป็นชื่อ Parent Project

                if (row[0] && (!row[1] || row[1].trim() === "") && (!row[2] || row[2].trim() === "")) {

                    currentProject = row[0].trim();

                    continue;

                }


                // สกัดข้อมูลแต่ละคอลัมน์ (รองรับโครงสร้างแบบไฟล์ตัวอย่าง)

                // Col 0: อาจจะว่างหรือเป็น BU, Col 1: Task/BU, Col 2: Detail...

                let bu = (row[1] && row[1].trim() !== "") ? row[1].trim() : (row[0] ? row[0].trim() : "");

                let task = row[2] ? row[2].trim() : "";

                let detail = row[3] ? row[3].trim() : "";

                let statusRaw = row[4] ? row[4].trim() : "";

                let nextStep = row[5] ? row[5].trim() : "";


                // ถ้าข้อมูลทุกช่องว่างหมด ให้ข้าม

                if (!bu && !task && !detail) continue;


                // วิเคราะห์สถานะจาก Text หรือ Emoji

                let statusIcon = "🟡";

                let statusText = "กำลังดำเนินการ";

                

                const fullText = row.join("").toLowerCase();

                if (fullText.includes("🟢") || fullText.includes("done") || statusRaw.includes("เสร็จ")) {

                    statusIcon = "🟢";

                    statusText = "เสร็จสิ้น / ตามแผน";

                }


                parsedData.push({

                    id: idCounter++,

                    parentProject: currentProject,

                    bu: bu,

                    task: task,

                    detail: detail,

                    statusIcon: statusIcon,

                    statusText: statusText,

                    nextStep: nextStep

                });

            }


            if (parsedData.length > 0) {

                rawData = parsedData;

                activeTab = "TV_VIEW"; // กลับไปหน้า TV View อัตโนมัติเมื่อโหลดเสร็จ

                renderApp();

            } else {

                alert("ไม่พบข้อมูลโปรเจกต์ที่ใช้งานได้ในไฟล์นี้");

            }

        }

    </script>
